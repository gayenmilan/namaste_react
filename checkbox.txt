<script>
document.getElementById('myButton').addEventListener('click', function() {
    // Get the IDs of the div elements
    var divIds = ['div1', 'div2', 'div3'];
    
    // Store the IDs in session storage
    sessionStorage.setItem('divIds', JSON.stringify(divIds));
    
    alert('Div IDs stored in session storage!');
});
